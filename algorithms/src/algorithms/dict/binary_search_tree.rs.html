  2 use std::fmt::Debug;

  5 // Priority Queue
  6 #[derive(Clone, Debug)]
  7 pub struct BinarySearchTree<K, V> {
  8     key: K,
  9     value: V,
 10     left: Option<Box<BinarySearchTree<K, V>>>,
 11     right: Option<Box<BinarySearchTree<K, V>>>,
 12     parent: Option<Box<BinarySearchTree<K, V>>>,
 13 }
 14 
 15 impl<K, V> From<(K, V)> for BinarySearchTree<K, V> {
 16     fn from(value: (K, V)) -> Self {
 17         Self {
 18             key: value.0,
 19             value: value.1,
 20             left: None,
 21             right: None,
 22             parent: None,
 23         }
 24     }
 25 }
 26 
 27 impl<K: Clone + Ord, V: Clone> From<Vec<(K, V)>> for BinarySearchTree<K, V> {
 28     fn from(value: Vec<(K, V)>) -> Self {
 29         let nodes: Vec<BinarySearchTree<K, V>> =
 30             value.into_iter().map(BinarySearchTree::from).collect();
 31 
 32         let mut root = nodes.get(0).unwrap().to_owned();
 33         for node in nodes.iter().skip(1) {
 34             root.insert(node.clone());
 35         }
 36 
 37         root
 38     }
 39 }
 40 
 41 impl<K: Ord, V> BinarySearchTree<K, V> {
 42     pub fn search(&self, key: K) -> Option<&V> {
 43         match key.cmp(&self.key) {
 44             Equal => Some(&self.value),
 45             Less => self.left.as_ref().and_then(|l| l.search(key)),
 46             Greater => self.right.as_ref().and_then(|r| r.search(key)),
 47         }
 48     }
 49 
 50     // Option<V>
 51     pub fn min(&self) -> (&K, &V) {
 52         self.left
 53             .as_ref()
 54             .and_then(|l| Some(l.min()))
 55             .unwrap_or((&self.key, &self.value))
 56     }
 57 
 58     pub fn max(&self) -> (&K, &V) {
 59         self.right
 60             .as_ref()
 61             .and_then(|r| Some(r.max()))
 62             .unwrap_or((&self.key, &self.value))
 63     }
 64 
 65     pub fn predecessor(&self) {
 66         match &self.left {
 67             Some(l) => (), // Most right in left
 68             None => (),    // First left parent
 69         }
 70     }
 71 
 72     pub fn successor() {}
 73 
 74     pub fn insert(&mut self, tree: BinarySearchTree<K, V>) {
 75         if tree.key < self.key {
 76             if let Some(l) = &mut self.left {
 77                 l.insert(tree);
 78             } else {
 79                 self.left = Some(Box::new(tree));
 80             }
 81         } else {
 82             if let Some(r) = &mut self.right {
 83                 r.insert(tree)
 84             } else {
 85                 // tree.parent = Some(Box::new(self));
 86                 self.right = Some(Box::new(tree));
 87                 self.right.unwrap().parent = Some(Box::new(self));
 88             }
 89         }
 90     }
 91 
 92     pub fn delete() {
 93         // No child
 94         // Only left
 95         // Only right
 96     }
 97 }
 98 
 99 impl<K, V> Index<K> for BinarySearchTree<K, V> {
100     type Output = V;
101     // type Idx = K;
102 
103     fn index(&self, index: K) -> &Self::Output {
104         &self.value
105     }
106 }
